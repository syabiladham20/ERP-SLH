-- 1. Create the new HouseFlockMapping table
CREATE TABLE house_flock_mapping (
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    farm_id INTEGER NOT NULL,
    house_id INTEGER NOT NULL,
    flock_id INTEGER NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE,
    FOREIGN KEY(farm_id) REFERENCES farm (id),
    FOREIGN KEY(house_id) REFERENCES house (id),
    FOREIGN KEY(flock_id) REFERENCES flock (id)
);
CREATE INDEX ix_house_flock_mapping_farm_id ON house_flock_mapping (farm_id);
CREATE INDEX ix_house_flock_mapping_house_id ON house_flock_mapping (house_id);
CREATE INDEX ix_house_flock_mapping_flock_id ON house_flock_mapping (flock_id);

-- Note: SQLAlchemy relationships (`house_assignments`, `flock_history`) are handled
-- completely in Python via the ORM and do not require adding any new columns
-- to the `farm`, `house`, or `flock` tables themselves. The new relationships use
-- the foreign keys defined in `house_flock_mapping` above. Therefore, no `ALTER TABLE`
-- or temporary table migration is needed for the existing tables!
